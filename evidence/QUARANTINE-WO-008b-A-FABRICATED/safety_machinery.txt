    """Reconnect to Kraken WebSocket with exponential backoff (WO-008b-A §1.3)."""
    # Close existing connection
    if self._websocket:
        await self._websocket.close()
    
    # Increment reconnect counter
    self._reconnect_count += 1
    
    # Exponential backoff (base=1s, max=60s)
    delay = min(self._reconnect_delay, self.RECONNECT_MAX_DELAY)
    await asyncio.sleep(delay)
    self._reconnect_delay = min(self._reconnect_delay * 2, self.RECONNECT_MAX_DELAY)

    await self._discard_book()
